/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
public class Solution {
    public TreeNode sortedArrayToBST(int[] nums) {
        if(nums.length==0) return null;
        else{
            return sortedArrayToBSTSol(nums,0,nums.length-1);
        }
    }
    public TreeNode sortedArrayToBSTSol(int[] nums, int a, int b){
        if(a==b){
            TreeNode tn = new TreeNode(nums[a]);
            return tn;
        }
        else if(a>b)return null;
        else{
            int half = (a+b)/2;
            TreeNode tnHalf = new TreeNode(nums[half]);
            tnHalf.left = sortedArrayToBSTSol(nums,a,half-1);
            tnHalf.right = sortedArrayToBSTSol(nums,half+1,b);
            return tnHalf;
        }
    }
    
    
}
