/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
public class Solution {
	int visitedDepth = 0;
    public List<Integer> rightSideView(TreeNode root) {
        List<Integer> result = new ArrayList<Integer>();
        if(root==null) return result;
        else{
        	rightSideView(root,1,result);
        	return result;
        }    
    }
    public void rightSideView(TreeNode root, int currDepth, List<Integer> result){
    	if(root==null) return;
    	else{
    		if(currDepth> visitedDepth) { //CAUTION: don't write as if(currDepth>visitedDepth++)
    		    result.add(root.val);
    		    visitedDepth++;
    		}
    		currDepth++;
    		rightSideView(root.right,currDepth,result);
    		rightSideView(root.left,currDepth,result);
    	}
    }
    
}
