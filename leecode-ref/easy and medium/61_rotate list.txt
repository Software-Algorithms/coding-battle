/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
public class Solution {
    public ListNode rotateRight(ListNode head, int k) {
        if(head==null||k==0) return head;
        ListNode helper = head;
        int length = 0;
        while(helper!=null){
            length++;
            helper = helper.next;
        }
        int offset = length - k%length;
        if(offset==length) return head;
        helper = head;
        while(offset-1>0){
            helper = helper.next;
            offset--;
        }
        ListNode newHead = helper.next;
        helper.next = null;
        helper = newHead;
        while(helper!=null&&helper.next!=null) helper = helper.next;
        helper.next = head;
        return newHead;
    }
}
