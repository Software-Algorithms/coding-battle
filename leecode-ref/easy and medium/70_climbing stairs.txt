public class Solution {
    public int climbStairs(int n) {
        if(n<=0)return 0;
        else if(n==1)return 1;
        else if(n==2)return 2;
        else{
        int first = 1;
        int second = 2;
        // below is the iteration form of dynamic programming and use only constant space.
        for(int i = 3; i<=n;i++){ 
            int temp = first + second;
            first = second;
            second = temp;
        }
        return second;
        }
    }
}
