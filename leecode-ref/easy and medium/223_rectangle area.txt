public class Solution {
    public int computeArea(int A, int B, int C, int D, int E, int F, int G, int H) {
        int union = (C-A)*(D-B) + (G-E)*(H-F);
        if(C<=E||G<=A||B>=H||F>=D) return union;
        else return union-(Math.max(A,E)-Math.min(C,G))*(Math.max(B,F)-Math.min(D,H));
    }
}
