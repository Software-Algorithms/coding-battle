public class Solution {
    public boolean isPowerOfTwo(int n) {
        if(n==1) return true;
        else if(n==0) return false;
        while(n!=2){
            if(n%2!=0) return false;
            n /= 2;
        }
        return true;
    }
}

public class Solution {
    public boolean isPowerOfTwo(int n) {
        int checker = 1;
        int count = 0;
        for(int i = 0;i<31;i++){ //only check first 31 bits. The 32th bit is mark for negative and positive.
            if((checker&n)==checker) count++;
            checker <<= 1;
        }
        if((checker&n)==checker) return false; //when n is negative, return false;
        else if(count==1) return true;
        else return false;
        
    }
}
