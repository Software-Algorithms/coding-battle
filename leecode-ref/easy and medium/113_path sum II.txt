/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
public class Solution {
    public List<List<Integer>> pathSum(TreeNode root, int sum) {
        List<List<Integer>> res = new ArrayList<List<Integer>>();
        List<Integer> curr = new ArrayList<Integer>();
        pathSum(root,sum,curr,0,res);
        return res;
    }
    public void pathSum(TreeNode root,int sum, List<Integer> curr, int currSum, List<List<Integer>> res){
        if(root==null) return;
        else if(root.left==null&&root.right==null&&(root.val+currSum==sum)){
            List<Integer> path = new ArrayList<Integer>(curr);
            path.add(root.val);
            res.add(path);
            return;
        }
        else{
            curr.add(root.val);
            pathSum(root.left,sum,curr,currSum+root.val,res);
            pathSum(root.right,sum,curr,currSum+root.val,res);
            curr.remove(curr.size()-1);
        }
    }
}
