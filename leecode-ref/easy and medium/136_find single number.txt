class Solution {
    public static int singleNumber(int[] nums) {
        int xor = 0;
        for(int i:nums){
            xor ^= i;
        }
        return xor;
    }
}

public class findsinglenum {

	public static void main(String[] args) {
        int[] testnums = {1,1,3,4,3,6,5,5,2,2,4};
        int single = Solution.singleNumber(testnums);
        System.out.println(single);
	}

}
