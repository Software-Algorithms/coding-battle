public class Solution {
    public boolean search(int[] nums, int target) {
        
        int a = 0, b = nums.length-1;
        while(a<=b){  // use a<=b in case ony 1 element in nums. with line 8 we will never have the case a = b later.
            int half = (a+b)/2;
            if(nums[half]==target) return true;
            else if(half==a){
                if(nums[b]==target) return true;
                else return false;
            }
            else if(nums[half]==nums[a]&&nums[half]==nums[b]){
                a = a +1;
                b = b -1;
            }
            else if(nums[half]>nums[a]){
                if(target>=nums[a]&&target<nums[half]) b = half-1;
                else a= half +1;
            }
            else{
                if(nums[half]==nums[a]) a = half +1; // or we can remove this sentence and in line 16 use >= to deal with value of half=left
                else if(target>nums[half]&&target<=nums[b]) a = half +1;
                else b = half -1;
            }
        }
        return false;
    }
}

public boolean search(int[] nums, int target) {
        if(nums[0]==target) return true;
        else if(nums.length == 1) return false;
        else{
            if(nums[0]<nums[nums.length-1]) return (biSearch(nums,0,nums.length-1,target)!=-1);
            else{
                int breakPoint = findMin_DQ(nums, 0 ,nums.length-1);
                if (breakPoint==0) return false;
                else if(nums[breakPoint]==target) return true;
                else if(nums[0]<=target&&nums[breakPoint-1]>=target) return (biSearch(nums,0,breakPoint-1,target)!=-1);
                return (biSearch(nums, breakPoint, nums.length-1, target)!=-1);
                
            }
        }
    }
    
    public int findMin_DQ(int[] nums,int a, int b){
        if(a==b)return a;
        else{
            int half = (a+b)/2;
            if(nums[half]>nums[half+1]) return half+1;
            else{
                if(nums[half]>nums[b]) return findMin_DQ(nums,half+1,b);
                else if(nums[half]==nums[b]){
                    if(nums[b]<nums[b-1]) return b; // now we are not looking for the min, but for the breakPoint,
                                                    // so for [1 1 3 1] we need find index 3 instead of 0;
                    return findMin_DQ(nums,a,b-1); //when half equals tail, can't make decision in which half part
                                                                            //the min is, so just strip the tail.
                } 
                else return findMin_DQ(nums,a,half);
            }
        }
    }
    
    public int biSearch(int[] nums, int a, int b, int target){
        if(a>b) return -1;
        else if(a==b){
            if(nums[a] == target) return a;
            else return -1;
        }
        else{
            int half = (a+b)/2;
            if(nums[half] == target) return half;
            else if (nums[half] > target) return biSearch(nums, a, half-1, target);
            else return biSearch(nums, half+1, b, target);
        }
    }
}
