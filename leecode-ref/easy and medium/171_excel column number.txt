class Solution {
    public int titleToNumber(String s) {
        if(s==null){
            return 0;
        }
        else
        {
        	int num = 0;
        	int multiplier = 1;
        	 for (int i = s.length()-1; i >=0 ; i--) {
                 char  item =  s.charAt(i);
                 int j = item - 64;
                 num += j*multiplier;
                 multiplier *= 26;
             }
        	 return num;
         }
    }
}
