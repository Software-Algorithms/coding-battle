public class Solution {
    public int removeDuplicates(int[] nums) {
        if(nums.length<=2) return nums.length;
        else{
            int curr = 2;
            for(int i = 2;i<nums.length;i++){
                if(nums[i]!=nums[curr-1]||nums[i]!=nums[curr-2]){
                    nums[curr] = nums[i];
                    curr++;
                }
            }
            return curr;
        }
        
    }
}
